Authentication

OAuth 2.0

Secure authentication using OAuth 2.0 with PKCE for public clients.

GET /oauth/authorize
  ?client_id=YOUR_CLIENT_ID
  &redirect_uri=YOUR_REDIRECT_URI
  &response_type=code
  &scope=tax_calculation document_upload
  &state=random_string

API Keys

For server-to-server communication, use API keys in request headers.

Authorization: Bearer YOUR_API_KEY
Content-Type: application/json

API Endpoints

Tax Calculation

POST /api/v1/calculate/income-tax

Calculate income tax liability based on income and deductions.

Request Body

{
  "financial_year": "2024-25",
  "income": {
    "salary": 800000,
    "business": 0,
    "capital_gains": 50000,
    "other_sources": 10000
  },
  "deductions": {
    "section_80c": 150000,
    "section_80d": 25000,
    "section_80g": 10000
  },
  "tax_regime": "new"
}

Response

{
  "success": true,
  "data": {
    "total_income": 860000,
    "taxable_income": 675000,
    "total_tax": 45000,
    "cess": 1800,
    "total_tax_liability": 46800,
    "effective_tax_rate": 5.44
  }
}
POST /api/v1/calculate/gst

Calculate GST for goods and services.

Request Body

{
  "base_amount": 10000,
  "gst_rate": 18,
  "is_interstate": false
}

Code Examples

Complete Tax Filing Example

from myeca import MyeCA
import json

# Initialize client
client = MyeCA(api_key="your_api_key")

# Calculate tax
tax_data = {
    "financial_year": "2024-25",
    "income": {
        "salary": 800000,
        "other_sources": 10000
    },
    "deductions": {
        "section_80c": 150000
    }
}

result = client.calculate_income_tax(tax_data)
print(f"Tax Liability: ‚Çπ{result['total_tax_liability']}")

# Upload documents
with open("form16.pdf", "rb") as f:
    doc = client.upload_document(
        file=f,
        document_type="form_16",
        financial_year="2024-25"
    )

# Create filing
filing = client.create_filing(
    user_id="user_123",
    financial_year="2024-25",
    documents=[doc['id']]
)

print(f"Filing ID: {filing['id']}")

Node.js Integration

const { MyeCA } = require('@myeca/sdk');

const client = new MyeCA('your_api_key');

async function fileTaxes() {
    try {
        // Calculate tax
        const taxResult = await client.calculateIncomeTax({
            financial_year: '2024-25',
            income: {
                salary: 800000,
                other_sources: 10000
            },
            deductions: {
                section_80c: 150000
            }
        });

        console.log(`Tax: ‚Çπ${taxResult.total_tax_liability}`);

        // Upload document
        const fs = require('fs');
        const document = await client.uploadDocument({
            file: fs.createReadStream('form16.pdf'),
            document_type: 'form_16',
            financial_year: '2024-25'
        });

        // Create filing
        const filing = await client.createFiling({
            user_id: 'user_123',
            financial_year: '2024-25',
            documents: [document.id]
        });

        console.log(`Filing created: ${filing.id}`);
    } catch (error) {
        console.error('Error:', error.message);
    }
}

fileTaxes();

REST API Calls

# Calculate Income Tax
curl -X POST https://api.myeca.in/v1/calculate/income-tax \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "financial_year": "2024-25",
    "income": {
      "salary": 800000,
      "other_sources": 10000
    },
    "deductions": {
      "section_80c": 150000
    }
  }'

# Upload Document
curl -X POST https://api.myeca.in/v1/documents/upload \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -F "file=@form16.pdf" \
  -F "document_type=form_16" \
  -F "financial_year=2024-25"

# Create Filing
curl -X POST https://api.myeca.in/v1/filing/create \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "user_id": "user_123",
    "financial_year": "2024-25",
    "filing_type": "individual"
  }'
